#Standardize data
def standardize_data(values):
    '''
    Fill in this function to return a standardized version of the given values,
    which will be in a NumPy array. Each value should be translated into the
    number of standard deviations that value is away from the mean of the data.
    (A positive number indicates a value higher than the mean, and a negative
    number indicates a value lower than the mean.)
    '''
    mean = values.mean()
    dev = values.std(ddof=0)
    return (values-mean)/dev

#Correlation
def correlation(x, y):
    """
    Pull the means and the standard deviations and then use this math rule for the Pearson's coefficient
    correlation = average of (x in standard units) times (y in standard units)
    
    """
    xmean = x.mean()
    ymean = y.mean()
    
    xstd = x.std(ddof=0)
    ystd = y.std(ddof=0)
    
    xstandard = ((x-xmean)/xstd)
    ystandard = ((y-ymean)/ystd)
    
    cor = (xstandard*ystandard).mean()
    
    return cor

#Pearson's and Scatterplots
def rAndGraph(l1,l2):
    """
    First, pull the names of the columns using the name propery. Replace any weird characters.
    Get the Pearson's coeffiecient from earlier function
    Prep output for ease of maintenance.
    Apply the logic and add the prefab verbage
    Graph
    """
    n1 = l1.name.replace("_", " ").title()
    n2 = l2.name.replace("_", " ").title()
    c = round(correlation(l1,l2),3)*100
    prefab1 = "As far as " + n1 + " and " + n2 + ", there is "
    prefab2 = "correlation with a value of " + str(c) + "%. Here's the graph: "
    
    if -25 < c < 25:
        print(prefab1 + "nearly no " + prefab2)
    elif (-50 < c <= -25) or (25 <= c < 50):
        print(prefab1 + "a slight " + prefab2)
    elif (-75 < c <= -50) or (50 <= c < 75):
        print(prefab1 + "some " + prefab2)
    else:
        print(prefab1 + "strong " + prefab2)
    
    plt.scatter(standardize_data(l1),standardize_data(l2))
    plt.title("Comparing " + n1 + " to " + n2 + " --- Please note: the data is standardized.")
    plt.xlabel(n1)
    plt.ylabel(n2)
    plt.show()

    
def titleAndDescribe(l1):

    Throw a title and parse the description for ease of reading

import matplotlib.mlab as mlab

def bellCurve(l1):
    mean = l1.mean()
    std = l1.std()
    x = np.linspace(mean - 4*std, mean + 4*std, 100)
    plt.plot(x,mlab.normpdf(x, mean, std),label=l1.name.title())
    plt.ylabel('Expectation')
    plt.xlabel('Units')
    plt.title(l1.name.title() + " and its Cool Looking Bell Curve! :D :D :D")
    plt.show()
